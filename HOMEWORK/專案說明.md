    width: 200px;
    height: 200px;
    -webkit-transition: width 1s, height 1s, -webkit-transform 1s; 
    transition: width 1s, height 1s, transform 1s;
    
}	
img:hover {
    width: 300px;
    height: 300px;
    -webkit-transform: rotate(360deg);
    transform: rotate(360deg);
}
